% CGEL Fragments — shared preamble
%
% Pulls in the portfolio house-style preamble, then adds handout-specific
% formatting (compact margins, tight spacing, numbered example/definition
% style) modelled on Pullum's Edinburgh "Advanced Topics in Syntax"
% handouts.
%
% The house-style preamble already loads geometry and langsci-gb4e; we
% override margins via \geometry{...} and use \ea/\z for numbered
% examples (matching house style).

\input{../.house-style/preamble.tex}

\geometry{a4paper,margin=2cm,top=1.8cm,bottom=2cm}

\usepackage{multicol}
\usepackage{amsmath,amssymb,amsthm}
\usepackage{enumitem}
\usepackage{tikz}
\usetikzlibrary{trees,positioning,fit}
\usepackage{forest}
\usepackage{titlesec}
\usepackage{fancyhdr}

% Tighter section headers, inline-bold style (Pullum-handout flavour)
\titleformat{\section}{\bfseries\large}{\thesection}{0.5em}{}
\titleformat{\subsection}[runin]{\bfseries}{\thesubsection}{0.5em}{}[\quad]
\titlespacing*{\section}{0pt}{1.5ex}{0.8ex}
\titlespacing*{\subsection}{0pt}{0.5ex}{0.5em}

% Definition shorthand
\newcommand{\defitem}[2]{\par\noindent\textbf{#1.}\ #2\par}

% Per-fragment identity (\renewcommand these in each fragment after
% \input{../shared/preamble.tex}).
\providecommand{\fragmentnumber}{Fragment N}
\providecommand{\fragmenttitle}{Untitled}
\providecommand{\fragmentdate}{\today}

% Handout header/footer
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\fancyhead[L]{\small\itshape CGEL Fragments\,/\,Reynolds}
\fancyhead[R]{\small\itshape \fragmentnumber\,---\,\fragmentdate}
\fancyfoot[C]{\thepage}

% Fragment title block
\newcommand{\fragmenthead}{%
  \begin{center}
    {\large\bfseries \fragmenttitle}\\[0.3em]
    {\small \fragmentnumber\,---\,\fragmentdate}
  \end{center}
  \vspace{0.5em}
}
